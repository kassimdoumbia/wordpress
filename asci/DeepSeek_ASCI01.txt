        // Navigation entre les sections
        document.querySelectorAll('.nav-link').forEach(link => {
            link.addEventListener('click', function(e) {
                e.preventDefault();
                
                // Masquer toutes les sections
                document.querySelectorAll('.section').forEach(section => {
                    section.classList.add('section-hidden');
                });
                
                // Afficher la section correspondante
                const targetSection = this.getAttribute('data-section');
                document.getElementById(targetSection).classList.remove('section-hidden');
                
                // Mettre à jour la classe active
                document.querySelectorAll('.nav-link').forEach(link => {
                    link.classList.remove('active');
                });
                this.classList.add('active');
            });
        });
        
        // Menu mobile

Ce code crée un site web moderne et responsive pour l'association ASCI avec les caractéristiques demandées :

1. Design adapté aux sites militaires avec une palette de couleurs bleu, blanc, rouge
2. Structure sémantique avec header, main, footer et sections
3. Logo au centre de la page d'accueil avec le texte au-dessus
4. Header et footer occupant chacun 1/5 de la page
5. Arrière-plan avec le drapeau français
6. Espace prévu pour l'intégration de Google Maps
7. Liens vers les réseaux sociaux dans le footer
8. Sections pour toutes les pages demandées
9. Galerie de photos et formulaire d'adhésion
10. Section Histoire avec toutes les dates commémoratives

Le site est entièrement responsive et s'adaptera à tous les appareils. Pour une utilisation réelle, il faudrait remplacer les placeholders par de vraies images et intégrer l'API Google Maps.
